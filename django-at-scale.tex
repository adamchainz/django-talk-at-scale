# project/admin.py
class ModelAdmin(admin.ModelAdmin):
    def queryset(self, request):
        qs = super(MySuperDuperAdmin, self) \
             .queryset(request)
        qs = qs._clone(klass=ApproxCountQuerySet)
        return qs
